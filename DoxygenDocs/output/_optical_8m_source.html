851  function cutPoly = occlusionCut(x,y,numCubesats) {
852 
853  method1Success = true;
854 
855  /* % concavity finding stuff
856  * find convex hull */
857  k = convhull(x,y);
858  convHull_poly = polyshape(x(k),y(k));
859 
860  /* calculate distance from boundary to convex hull */
861  [xMesh,yMesh] = meshgrid(1:max(x)+2,1:max(y)+2);
862  xMesh = xMesh - 1;
863  yMesh = yMesh - 1;
864  convHullOuterBin = ~isinterior(convHull_poly,xMesh(:),yMesh(:));
865  convHullOuterBin = reshape(convHullOuterBin,[max(y)+2,max(x)+2]);
866  D = bwdist(convHullOuterBin);
867  r = zeros(numel(x),1);
868  for i = 1:numel(x)
869  r(i) = D(y(i),x(i));
870  if r(i) < 1
871  r(i) = 0;
872  end
873  end
874 
875  /* estimate number of concavity points */
876  windowSize = ceil(numel(r)*0.05);
877  b = (1/windowSize)*ones(1,windowSize);
878  a = 1;
879  r_filter = filter(b,a,r);
880  minProm = 0.1;
881  [tmpPks,fitPks,~,tmpProm] = findpeaks(r_filter," SortStr "," descend ");
882  [tmpPks,fitPks,~,tmpProm] = findpeaks(r," SortStr "," descend ",...
883  " MinPeakProminence ",max(tmpProm.*0.05),...
884  " MinPeakWidth ",0,...
885  " MaxPeakWidth ",Inf,...
886  " MinPeakHeight ",0,...
887  " MinPeakDistance ",numel(r_filter)*0.25);

895  end

918
