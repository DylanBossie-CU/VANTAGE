410  function pos = SensorFusion(camOrigin,camVec,pos_TOF) {
411 
412  /* Make everything a column vector */
413  if numel(camOrigin) > size(camOrigin,1)
414  camOrigin = camOrigin^t;
415  end
416  if numel(camVec) > size(camVec,1)
417  camVec = camVec^t;
418  end
419  if numel(pos_TOF) > size(pos_TOF,1)
420  pos_TOF = pos_TOF^t;
421  end
422 
423  /* Normalize camera vector */
424  camVec = camVec./norm(camVec);
425 
426  /* Define line vector */
427  a = camOrigin - camVec;
428 
429  /* Define vector from line to TOF point */
430  b = pos_TOF - camVec;
431 
432  /* Calculate vector cross product */
433  vec = cross(a,b);
434 
435  /* Calculate cam and TOF weights */
436  sig_TOF = this.TOF.TofWeighting(norm(pos_TOF));
437  sig_cam = this.Optical.OpticalWeighting(norm(pos_TOF));
438 
439  /* Calculate distance along line for weigthed center point */
440  d = norm(vec) / norm(a);
441  if sig_TOF~=0
442  q = sig_TOF./(sig_cam+sig_TOF);
443  else
444  q = 0;
445  end
446  d_q = d.*q;
447 
448  /* Calculate position for camera */
449  mag_TOF = norm(pos_TOF);
450  mag_cam = sqrt(mag_TOF.^2 - d.^2);
451  pos_cam = camOrigin + mag_cam.*camVec;
452 
453  /* Calculate error-weighted centroid position */
454  posDir = pos_cam - pos_TOF; posDir = posDir./norm(posDir);
455  pos = pos_TOF + posDir.*d_q;
456  }
