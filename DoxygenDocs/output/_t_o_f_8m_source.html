681  function [this , CubeSats_TOF ] = cubesatPointsFromPC(pc,CubeSats_TOF) {
682  z = sort(pc.Location(:,3));
683 
684  /* Start the loop */
685  NPeaks = Inf;
686  bw = 0.01;
687  counter = 0;
688  while NPeaks > length(CubeSats_TOF)-1
689  /* Calculate k-squares density */
690  [zDense,zBin] = ksdensity(z," bandwidth ",bw," function "," pdf ");
691  /* Identify split locations */
692  c = 0.3;
693  h = -1.5;
694  warning(" off "," signal:findpeaks:largeMinPeakHeight ")
695  [pks,locs] = findpeaks(-zDense,zBin," MinPeakProminence ",c," MinPeakHeight ",h);
696  warning(" on "," signal:findpeaks:largeMinPeakHeight ")
697  /* Update bandwidth */
698  NPeaks = length(locs);
699  bw = 1.05*bw;
700  /* give up if too many tries */
701  counter = counter + 1;
702  if counter > 30
703  /* warning('Bandwidth for point splitting ksdensity function did not converge in 30 tries') */
704  nsplit = 0;
705  NPeaks = [];
706  locs = [];
707  break
708  end
709  end
710  if this.showDebugPlots /* useful for debugging */
711 
712  figure
713  hold on
714  grid on
715  grid minor
716  plot(zBin,-zDense);
717  plot(locs,pks," r* ")
718  xlabel(" z (m) ")
719  ylabel(" Percent point density ")
720  hold off
721  end
722  locs = flip(locs);
723  nSplit = numel(locs);
724  /* Separate point cloud by split planes */
725  for i = 1:nSplit+1
726  /* Find indices of the current cubesat points in the point
727  * cloud */
728  if i == 1
729  if nSplit > 0
730  I = pc.Location(:,3)>=locs(i);
731  else
732  I = ~isnan(pc.Location(:,3));
733  end
734  elseif i == nSplit+1
735  I = pc.Location(:,3)<locs(nSplit);
736  else
737  I = (pc.Location(:,3)>=locs(i)) & (pc.Location(:,3)<locs(i-1));
738  end
739  /* Skip this CubeSat if it's too close to the edge of the
740  * FOV */
741  maxZfov = max(...
742  max(abs(pc.Location(I,1)))/tand(this.HVfov_deg(1)/2),...
743  max(abs(pc.Location(I,2)))/tand(this.HVfov_deg(2)/2)...
744  );
745  satPcZDistFromFOV = min(pc.Location(I,3)-abs(pc.Location(I,1))./tand(this.HVfov_deg(1)/2),...
746  pc.Location(I,3)-abs(pc.Location(I,2))./tand(this.HVfov_deg(2)/2));
747  /* skip if first percentile of data is less than two cm from
748  * the FOV --and-- if the 99th percentile of data is less
749  * than 20 cm from the highest point of the FOV directly
750  * below (-t3) the cubesat */
751  skipThisCubesat = (prctile(satPcZDistFromFOV,1) < 2/100) & ...
752  (prctile(pc.Location(I,3),99)-maxZfov < 15/100);
753  if skipThisCubesat
754  continue
755  end
756  CubeSats_TOF(i).pc = pointCloud(pc.Location(I,:));
757  end

793  planes = CubeSat_TOF.faces;
794 
795  /* Tracking of points not associated with any plane */
796  remainPtCloud = CubeSat_TOF.pc;
797 
798  /* %% Find planes until no more planes exist */
799  while remainPtCloud.Count > minPtsInPlane
800  /* %% Fit a plane to the ptCloud */
801  warning(" off "," vision:pointcloud:notEnoughInliers ")

803  warning(" on "," vision:pointcloud:notEnoughInliers ")
804

806  if isempty(inlierIndices) || numel(inlierIndices) < minPtsInPlane
807  break
808  else
809  numPlanes = numPlanes + 1; /* add a plane if one was found */
810 
811  end
812 
813  /* %% Extract plane from ptCloud
814  * extract */
815  plane = select(remainPtCloud,inlierIndices);
816  /* store in output */
817  planes(numPlanes).planeCloud = plane;
818 
819  /* %% Extract remaining points from ptCloud */
820  remainPtCloud = select(remainPtCloud,outlierIndices);
821 
822  /* %% Obtain planar basis */
823  [planes(numPlanes).n,planes(numPlanes).V,planes(numPlanes).o] = this.affine_fit(double(planes(numPlanes).planeCloud.Location));
824 
825  /* %% Remove if repeat plane (e.g. if this is the cubesat 'feet')
826  * For all previous planes */
827  for i = 1:numPlanes-1
828  /* if normal is within 45 deg of prior plane's normal or antinormal */
829  if acosd(dot(planes(numPlanes).n,planes(i).n)) < 45 || acosd(dot(planes(numPlanes).n,planes(i).n)) > 135
830  /* if this plane is smaller than previous plane */
831  if planes(numPlanes).planeCloud.Count <= planes(i).planeCloud.Count
832  /* delete this plane */
833  planes(numPlanes) = [];
834  numPlanes = numPlanes-1;
835  break
836  /* if the previous plane is smaller than this plane */
837  else
838  /* replace previous plane with this plane and delete this plane */
839  planes(i) = planes(numPlanes);
840  planes(numPlanes) = [];
841  numPlanes = numPlanes-1;
842  break
843  end
844  end
845  end
846 
847  /* %% Update minimum number points required to make a plane
848  * Take minPoints to be one-twentieth the number of points found in the first
849  * plane */
850  if numPlanes == 1 && minPtsUnset
851  minPtsInPlane = round(planes(numPlanes).planeCloud.Count/20);
852  minPtsUnset = false;
853  end
854  end
855 
856  /* %% Check that no more than three planes were identified */
857  if numPlanes > 3
858  /* error('fitPlanesToCubesat identified more than three planes for a single cubesat') */
859  CubeSat_TOF.trustedSat= 0;
860  end
861 
862  /* %% Check that planes are mutually orthogonal */
863  for i = 1:numPlanes
864  for j = 1:numPlanes
865  if abs(asind(dot(planes(i).n,planes(j).n))) > 8 && i~=j
866  /* CubeSat_TOF.trustedSat = 0;
867  *warning('fitPlanesToCubesat identified planes which are not mutually orthogonal by greater than 8 deg') */
868  end
869  end
870  end
871 
872  /* %% Package planes and numPlanes into CubeSat object */
873  CubeSat_TOF.faces= planes;
874  CubeSat_TOF.numVisibleFaces= numPlanes;
875

937  function CubeSat_TOF = centroid1(CubeSat_TOF) {
938  [CubeSat_TOF.faces] = this.getFaceCentroidAndCorners(CubeSat_TOF.faces);
939 
940  /* %% Identify face corners using bounding box of face */
941  [CubeSat_TOF] = this.diagnoseFace(CubeSat_TOF);
942 
943  /* %% Extract data */
944  face = CubeSat_TOF.faces;
945 
946  /* %% Find centroid of face in 3D */
947  if face.fullFace
948  /* mean corners to find centroid */
949  centr = mean(face.corners,2);
950  else
951  /* project to centroid from two most distant (downrange) corners
952  * Obtain distant two corners and their midPt */
953  [~,I] = sort(face.corners(3,:)," descend ");
954  farCorners = face.corners(:,I(1:2));
955  farMidPt = mean(farCorners,2);
956 
957  /* Determine which trustedLen corresponds to the side that
958  * is orthogonal to corners */
959  [~,I] = min(abs(face.trustedLen-norm(farCorners(:,1)-farCorners(:,2))));
960  projLength = face.trustedLen([1:I-1,I+1:end]);
961 
962  /* Extract face normal vector */
963  nhat = face(1).n;
964 
965  /* Obtain vector from midPt to a corner */
966  relCorner = farCorners(:,2) - farMidPt;
967 
968  /* Obtain projection direction from farMidPt orthogonal to corners and
969  * towards the face interior */
970  midPtToInterior = face.faceCentr - farMidPt;
971  projHat = this.unitvec( cross(nhat,relCorner) );
972  projHat = sign( dot(projHat,midPtToInterior) ) * projHat;
973 
974  /* Project from farMidPt to the desired face centroid */
975  centr = farMidPt + projLength/2 * projHat;
976  end
977 
978  /* %% Project inward to volumetric centroid */
979  d = this.distInFromFace(CubeSat_TOF,face.faceIndex);
980  CubeSat_TOF.centroid_TCF= centr + d*sign(dot(centr,face.n))*face.n;
981  }
1009  function CubeSat_TOF = centroid2(CubeSat_TOF) {
1010  if rank([CubeSat_TOF.faces.n]) ~= 2
1011  error(" The two planes identified do not intersect as a line ")
1012  end
1013 
1014  /* %% Find faces' centroids and corners
1015  *%% and determine if each face qualifies as a fullFace */
1016  CubeSat_TOF.faces= this.getFaceCentroidAndCorners(CubeSat_TOF.faces);
1017  CubeSat_TOF = this.diagnoseFace(CubeSat_TOF);
1018 
1019  /* %% Extract data */
1020  faces = CubeSat_TOF.faces;
1021 
1022  /* %% Determine unit vector parallel to intersection */
1023  v_intersect = cross(faces(1).n,faces(2).n);
1024 
1025  /* %% Get mean point along intersect
1026  * Project trusted corners onto intersect */
1027  allTrustedCorners = [faces(1).trustedCorners,faces(2).trustedCorners];
1028  corners_alongIntersect = allTrustedCorners^t * v_intersect;
1029  /* Find distance from min to mean along intersect */
1030  [min_alongIntersect,minI] = min(corners_alongIntersect);
1031  tmp_mean = mean([max(corners_alongIntersect),min_alongIntersect]);
1032  mean_alongIntersect = tmp_mean-min_alongIntersect;
1033  /* Identify min_alongIntersect in 3D space by projecting min
1034  * point onto intersect three dimensionally */
1035  if minI <= 4 /* if point is in face 1 */
1036 
1037  otherFace = 2;
1038  else /* if point is in face 2 */
1039 
1040  otherFace = 1;
1041  end
1042  orthDist = (dot(faces(otherFace).n,faces(otherFace).o)-...
1043  dot(faces(otherFace).n,allTrustedCorners(:,minI)))/norm(faces(otherFace).n)^2;
1044  minpt = allTrustedCorners(:,minI) + orthDist*faces(otherFace).n;
1045  /* Locate mean point along intersect in 3D */
1046  meanpt = minpt + mean_alongIntersect*v_intersect;
1047 
1048  /* %% Project into centroid from mean point along intersect */
1049  n = zeros(3,2);
1050  for i = 1:2
1051  /* ensure that unit normals are inward facing */
1052  n(:,i) = sign(dot(meanpt,faces(i).n))*faces(i).n;
1053  /* calculate orthogonal distance from each face to the centroid */
1054  d(i) = this.distInFromFace(CubeSat_TOF,faces(i).faceIndex);
1055  end
1056  /* Project to centroid */
1057  CubeSat_TOF.centroid_TCF= meanpt + sum(d.*n,2);
1058  }
1086  function CubeSat_TOF = centroid3(CubeSat_TOF) {
1087  CubeSat_TOF.faces= this.getFaceCentroidAndCorners(CubeSat_TOF.faces);
1088  CubeSat_TOF = this.diagnoseFace(CubeSat_TOF);
1089 
1090  /* %% Extract data */
1091  faces = CubeSat_TOF.faces;
1092 
1093  /* %% Find plane intersection */
1094  A = zeros(3);
1095  b = zeros(3,1);
1096  for i = 1:3
1097  A(i,1:3) = faces(i).n;
1098  b(i,1) = faces(i).n(1)*faces(i).o(1) + faces(i).n(2)*faces(i).o(2) + faces(i).n(3)*faces(i).o(3);
1099  end
1100  if rank(A) ~= 3
1101  error(" Planes do not intersect in the expected way ")
1102  end
1103  ptIntersect = (A\b);
1104 
1105  /* %% Project into centroid from intersection point */
1106  n = zeros(3,3);
1107  d = zeros(1,3);
1108  for i = 1:3
1109  /* ensure that unit normals are inward facing */
1110  n(:,i) = sign(dot(ptIntersect,faces(i).n))*faces(i).n;
1111  /* calculate orthogonal distance from each face to the centroid */
1112  d(i) = this.distInFromFace(CubeSat_TOF,faces(i).faceIndex);
1113  end
1114  /* Orthogonalize n */
1115  [Q,R] = qr(n);
1116  n_orth = sign(diag(R))^t .* Q;
1117  /* Project to centroid */
1118  CubeSat_TOF.centroid_TCF= ptIntersect + sum(d.*n_orth,2);
1119  }
1147  function faces = getFaceCentroidAndCorners(faces) {
1148  for ii = 1:length(faces)
1149  face = faces(ii);
1150  /* %% Project points onto 2d plane */
1151  plane = face.planeCloud;
1152  inPlane = double( (plane.Location-face(1).o^t)*face.V );
1153 
1154  /* %% Find plane boundary points */
1155  I_bound = boundary(inPlane(:,1),inPlane(:,2));
1156 
1157  /* %% Find filtered boundary */
1158  boundaryPtsRaw = [inPlane(I_bound,1)" ;inPlane(I_bound,2) "];
1159  order = 2;
1160  frameLen = this.roundToNearestOdd(size(boundaryPtsRaw,2)/5);
1161  if frameLen <= 1
1162  boundaryPts = boundaryPtsRaw;
1163  else
1164  boundaryPts = sgolayfilt(boundaryPtsRaw,order,frameLen,[],2);
1165  end
1166 
1167  /* %% Find face centroid */
1168  faceShape = polyshape(inPlane(I_bound,1),inPlane(I_bound,2)," simplify ",false);
1169  [x,y] = centroid(faceShape);
1170  face.faceCentr= face.o + face.V * [x;y];
1171 
1172  /* Find face boundary in face plane */
1173  corners_inPlane = this.minBoundingBox(boundaryPts);
1174  /* Convert to 3D */
1175  face.corners= zeros(3,4);
1176  for i = 1:4
1177  face.corners(:,i) = face.o + face.V * corners_inPlane(:,i);
1178  end
1179 
1180  /* %% repackage into faces */
1181  faces(ii) = face;
1182  end
1183  }
1204  function CubeSat_TOF = diagnoseFace(CubeSat_TOF) {
1205  for ii = 1:length(CubeSat_TOF.faces)
1206  /* Extract data */
1207  face = CubeSat_TOF.faces(ii);
1208  corners = face.corners;
1209  /* Initialize boolean */
1210  face.fullFace= 1;
1211  /* %% Decide if full face is present or not
1212  * Determine if bounding box is near cubesat size */
1213  len = zeros(1,2);
1214  deltaLen = zeros(3,2);
1215  faceMatches = cell(1,2);
1216  for i = 1:2
1217  len(i) = norm(corners(:,i+1) - corners(:,i));
1218  deltaLen(:,i) = abs(len(i) - CubeSat_TOF.actualDims);
1219  faceMatches[i] = find(deltaLen(:,i) < this.faceLenMatchTol);
1220  end
1221  /* For fullFace, both sides must have a length match */
1222  fullFaceCondition = ~isempty(faceMatches[1]) & ~isempty(faceMatches[2]);
1223  /* and each side must have an independent match */
1224  matchCombinations = combvec(faceMatches[1]" ,faceMatches{2} ");
1225  diffSides = ~(matchCombinations(1,:)-matchCombinations(2,:));
1226  fullFaceCondition = fullFaceCondition & any(diffSides);
1227 
1228  /* fullFaceCondition = fullFaceCondition & sum(ismember(faceMatches{1},faceMatches{2}))~=length(faceMatches{1});
1229  *fullFaceCondition = fullFaceCondition & sum(ismember(faceMatches{2},faceMatches{1}))~=length(faceMatches{2}); */
1230  if ~fullFaceCondition
1231  face.fullFace= 0;
1232  end
1233 
1234  /* %% Determine trusted lengths */
1235  if face.fullFace
1236  /* trusted side lengths are visible side lengths */
1237  face.trustedLen= len;
1238  /* trusted corners are identified corners */
1239  face.trustedCorners= corners;
1240  else
1241  /* NOTE: this assumes that the CubeSat has not tumbled much
1242  * since launching, which is very valid for the TOF camera
1243  * Determine trusted lengths */
1244  [trustedActualDimI,trustedLenI] = find(min(min(deltaLen))==deltaLen,1);
1245  face.trustedLen= zeros(1,2);
1246  face.trustedLen(1) = len(trustedLenI);
1247  if trustedActualDimI ~= find(CubeSat_TOF.actualDims == max(CubeSat_TOF.actualDims))
1248  face.trustedLen(2) = max(CubeSat_TOF.actualDims);
1249  else
1250  face.trustedLen(2) = min(CubeSat_TOF.actualDims);
1251  end
1252  /* %% Determine trusted corners
1253  * project to centroid from two most distant (downrange) corners
1254  * Obtain distant two corners and their midPt */
1255  if (mean(corners(3,:)) > 3) || (CubeSat_TOF.rangeOrder == 1)
1256  sortDirection = " ascend ";
1257  else
1258  sortDirection = " descend ";
1259  end
1260  [~,I] = sort(face.corners(3,:)," descend ");
1261  farCorners = face.corners(:,I(1:2));
1262  nearBadCornersUnordered = face.corners(:,I(3:4));
1263 
1264  /* Determine which trustedLen corresponds to the side that
1265  * is orthogonal to corners */
1266  [~,I] = min(abs(face.trustedLen-norm(farCorners(:,1)-farCorners(:,2))));
1267  projLength = face.trustedLen([1:I-1,I+1:end]);
1268 
1269  /* Determine near bad corners which correspond to far corners */
1270  nearBadCorners = zeros(3,2);
1271  for i = 1:size(farCorners,2)
1272  for j = 1:size(nearBadCornersUnordered,2)
1273  v1 = farCorners(:,[1:i-1,i+1:end]) - farCorners(:,i);
1274  v2 = nearBadCornersUnordered(:,j) - farCorners(:,i);
1275  if abs(dot(v1,v2)) < 1e-06
1276  nearBadCorners(:,i) = nearBadCornersUnordered(:,j);
1277  end
1278  end
1279  end
1280 
1281  /* Project through near bad corners to near corners */
1282  nearCorners = zeros(3,2);
1283  for i = 1:2
1284  u = this.unitvec(nearBadCorners(:,i) - farCorners(:,i));
1285  nearCorners(:,i) = farCorners(:,i) + projLength*u;
1286  end
1287 
1288  /* Assemble corners in circular order */
1289  face.trustedCorners= [fliplr(nearCorners),farCorners];
1290  end
1291 
1292  /* %% Determine faceIndex
1293  * Dimensions to choose from, corresponding to indices of
1294  * CubeSat.actualDims */
1295  dims = 1:length(CubeSat_TOF.actualDims);
1296  /* Calculate standard areas */
1297  C = combnk(dims,2);
1298  Avec = prod(CubeSat_TOF.actualDims(C),2);
1299  /* Calculate difference between area and standard areas */
1300  faceArea = prod(face.trustedLen);
1301  dA = abs(faceArea-Avec);
1302  /* Choose nearest size */
1303  [~,face.faceIndex] = min(dA);
1304 
1305  /* %% Repackage into CubeSat */
1306  CubeSat_TOF.faces(ii) = face;
1307  end
1308 
1309  }

1539  function bb = minBoundingBox(X) {
1540 
1541  /* compute the convex hull (CH is a 2*k matrix subset of X) */
1542  k = convhull(X(1,:),X(2,:));
1543  CH = X(:,k);
1544 
1545  /* compute the angle to test, which are the angle of the CH edges as:
1546  * "one side of the bounding box contains an edge of the convex hull" */
