678  function [this , CubeSats_TOF ] = cubesatPointsFromPC(pc,CubeSats_TOF) {
679  z = sort(pc.Location(:,3));
680 
681  /* Start the loop */
682  NPeaks = Inf;
683  bw = 0.01;
684  counter = 0;
685  while NPeaks > length(CubeSats_TOF)-1
686  /* Calculate k-squares density */
687  [zDense,zBin] = ksdensity(z," bandwidth ",bw," function "," pdf ");
688  /* Identify split locations */
689  c = 0.3;
690  h = -1.5;
691  warning(" off "," signal:findpeaks:largeMinPeakHeight ")
692  [pks,locs] = findpeaks(-zDense,zBin," MinPeakProminence ",c," MinPeakHeight ",h);
693  warning(" on "," signal:findpeaks:largeMinPeakHeight ")
694  /* Update bandwidth */
695  NPeaks = length(locs);
696  bw = 1.05*bw;
697  /* give up if too many tries */
698  counter = counter + 1;
699  if counter > 30
700  /* warning('Bandwidth for point splitting ksdensity function did not converge in 30 tries') */
701  nsplit = 0;
702  NPeaks = [];
703  locs = [];
704  break
705  end
706  end
707  if this.showDebugPlots /* useful for debugging */
708 
709  figure
710  hold on
711  grid on
712  grid minor
713  plot(zBin,-zDense);
714  plot(locs,pks," r* ")
715  xlabel(" z (m) ")
716  ylabel(" Percent point density ")
717  hold off
718  end
719  locs = flip(locs);
720  nSplit = numel(locs);
721  /* Separate point cloud by split planes */
722  for i = 1:nSplit+1
723  /* Find indices of the current cubesat points in the point
724  * cloud */
725  if i == 1
726  if nSplit > 0
727  I = pc.Location(:,3)>=locs(i);
728  else
729  I = ~isnan(pc.Location(:,3));
730  end
731  elseif i == nSplit+1
732  I = pc.Location(:,3)<locs(nSplit);
733  else
734  I = (pc.Location(:,3)>=locs(i)) & (pc.Location(:,3)<locs(i-1));
735  end
736  /* Skip this CubeSat if it's too close to the edge of the
737  * FOV */
738  maxZfov = max(...
739  max(abs(pc.Location(I,1)))/tand(this.HVfov_deg(1)/2),...
740  max(abs(pc.Location(I,2)))/tand(this.HVfov_deg(2)/2)...
741  );
742  satPcZDistFromFOV = min(pc.Location(I,3)-abs(pc.Location(I,1))./tand(this.HVfov_deg(1)/2),...
743  pc.Location(I,3)-abs(pc.Location(I,2))./tand(this.HVfov_deg(2)/2));
744  /* skip if first percentile of data is less than two cm from
745  * the FOV --and-- if the 99th percentile of data is less
746  * than 20 cm from the highest point of the FOV directly
747  * below (-t3) the cubesat */
748  skipThisCubesat = (prctile(satPcZDistFromFOV,1) < 2/100) & ...
749  (prctile(pc.Location(I,3),99)-maxZfov < 15/100);
750  if skipThisCubesat
751  continue
752  end
753  CubeSats_TOF(i).pc = pointCloud(pc.Location(I,:));
754  end

795  /* %% Find planes until no more planes exist */
796  while remainPtCloud.Count > minPtsInPlane
797  /* %% Fit a plane to the ptCloud */
798  warning(" off "," vision:pointcloud:notEnoughInliers ")
799  [~,inlierIndices,outlierIndices] = pcfitplane(remainPtCloud,this.ptMaxDistFromPlane);
800  warning(" on "," vision:pointcloud:notEnoughInliers ")
801 
802  /* %% Quit now if no significant planes found */
803  if isempty(inlierIndices) || numel(inlierIndices) < minPtsInPlane
804  break
805  else
806  numPlanes = numPlanes + 1; /* add a plane if one was found */
807 
808  end
809 
810  /* %% Extract plane from ptCloud
811  * extract */
812  plane = select(remainPtCloud,inlierIndices);
813  /* store in output */
814  planes(numPlanes).planeCloud = plane;
815 
816  /* %% Extract remaining points from ptCloud */
817  remainPtCloud = select(remainPtCloud,outlierIndices);
818 
819  /* %% Obtain planar basis */
820  [planes(numPlanes).n,planes(numPlanes).V,planes(numPlanes).o] = this.affine_fit(double(planes(numPlanes).planeCloud.Location));
821 
822  /* %% Remove if repeat plane (e.g. if this is the cubesat 'feet')
823  * For all previous planes */
824  for i = 1:numPlanes-1
825  /* if normal is within 45 deg of prior plane's normal or antinormal */
826  if acosd(dot(planes(numPlanes).n,planes(i).n)) < 45 || acosd(dot(planes(numPlanes).n,planes(i).n)) > 135
827  /* if this plane is smaller than previous plane */
828  if planes(numPlanes).planeCloud.Count <= planes(i).planeCloud.Count
829  /* delete this plane */
830  planes(numPlanes) = [];
831  numPlanes = numPlanes-1;
832  break
833  /* if the previous plane is smaller than this plane */
834  else
835  /* replace previous plane with this plane and delete this plane */
836  planes(i) = planes(numPlanes);
837  planes(numPlanes) = [];
838  numPlanes = numPlanes-1;
839  break
840  end
841  end
842  end
843 
844  /* %% Update minimum number points required to make a plane
845  * Take minPoints to be one-twentieth the number of points found in the first
846  * plane */
847  if numPlanes == 1 && minPtsUnset
848  minPtsInPlane = round(planes(numPlanes).planeCloud.Count/20);
849  minPtsUnset = false;
850  end
851  end
852 
853  /* %% Check that no more than three planes were identified */
854  if numPlanes > 3
855  /* error('fitPlanesToCubesat identified more than three planes for a single cubesat') */
856  CubeSat_TOF.trustedSat= 0;
857  end
858 
859  /* %% Check that planes are mutually orthogonal */
860  for i = 1:numPlanes
861  for j = 1:numPlanes
862  if abs(asind(dot(planes(i).n,planes(j).n))) > 8 && i~=j
863  /* CubeSat_TOF.trustedSat = 0;
864  *warning('fitPlanesToCubesat identified planes which are not mutually orthogonal by greater than 8 deg') */
865  end
866  end
867  end
868 
869  /* %% Package planes and numPlanes into CubeSat object */
870  CubeSat_TOF.faces= planes;
871  CubeSat_TOF.numVisibleFaces= numPlanes;
872

934  function CubeSat_TOF = centroid1(CubeSat_TOF) {
935  [CubeSat_TOF.faces] = this.getFaceCentroidAndCorners(CubeSat_TOF.faces);
936 
937  /* %% Identify face corners using bounding box of face */
938  [CubeSat_TOF] = this.diagnoseFace(CubeSat_TOF);
939 
940  /* %% Extract data */
941  face = CubeSat_TOF.faces;
942 
943  /* %% Find centroid of face in 3D */
944  if face.fullFace
945  /* mean corners to find centroid */
946  centr = mean(face.corners,2);
947  else
948  /* project to centroid from two most distant (downrange) corners
949  * Obtain distant two corners and their midPt */
950  [~,I] = sort(face.corners(3,:)," descend ");
951  farCorners = face.corners(:,I(1:2));
952  farMidPt = mean(farCorners,2);
953 
954  /* Determine which trustedLen corresponds to the side that
955  * is orthogonal to corners */
956  [~,I] = min(abs(face.trustedLen-norm(farCorners(:,1)-farCorners(:,2))));
957  projLength = face.trustedLen([1:I-1,I+1:end]);
958 
959  /* Extract face normal vector */
960  nhat = face(1).n;
961 
962  /* Obtain vector from midPt to a corner */
963  relCorner = farCorners(:,2) - farMidPt;
964 
965  /* Obtain projection direction from farMidPt orthogonal to corners and
966  * towards the face interior */
967  midPtToInterior = face.faceCentr - farMidPt;
968  projHat = this.unitvec( cross(nhat,relCorner) );
969  projHat = sign( dot(projHat,midPtToInterior) ) * projHat;
970 
971  /* Project from farMidPt to the desired face centroid */
972  centr = farMidPt + projLength/2 * projHat;
973  end
974 
975  /* %% Project inward to volumetric centroid */
976  d = this.distInFromFace(CubeSat_TOF,face.faceIndex);
977  CubeSat_TOF.centroid_TCF= centr + d*sign(dot(centr,face.n))*face.n;
978  }
1006  function CubeSat_TOF = centroid2(CubeSat_TOF) {
1007  if rank([CubeSat_TOF.faces.n]) ~= 2
1008  error(" The two planes identified do not intersect as a line ")
1009  end
1010 
1011  /* %% Find faces' centroids and corners
1012  *%% and determine if each face qualifies as a fullFace */
1013  CubeSat_TOF.faces= this.getFaceCentroidAndCorners(CubeSat_TOF.faces);
1014  CubeSat_TOF = this.diagnoseFace(CubeSat_TOF);
1015 
1016  /* %% Extract data */
1017  faces = CubeSat_TOF.faces;
1018 
1019  /* %% Determine unit vector parallel to intersection */
1020  v_intersect = cross(faces(1).n,faces(2).n);
1021 
1022  /* %% Get mean point along intersect
1023  * Project trusted corners onto intersect */
1024  allTrustedCorners = [faces(1).trustedCorners,faces(2).trustedCorners];
1025  corners_alongIntersect = allTrustedCorners^t * v_intersect;
1026  /* Find distance from min to mean along intersect */
1027  [min_alongIntersect,minI] = min(corners_alongIntersect);
1028  tmp_mean = mean([max(corners_alongIntersect),min_alongIntersect]);
1029  mean_alongIntersect = tmp_mean-min_alongIntersect;
1030  /* Identify min_alongIntersect in 3D space by projecting min
1031  * point onto intersect three dimensionally */
1032  if minI <= 4 /* if point is in face 1 */
1033 
1034  otherFace = 2;
1035  else /* if point is in face 2 */
1036 
1037  otherFace = 1;
1038  end
1039  orthDist = (dot(faces(otherFace).n,faces(otherFace).o)-...
1040  dot(faces(otherFace).n,allTrustedCorners(:,minI)))/norm(faces(otherFace).n)^2;
1041  minpt = allTrustedCorners(:,minI) + orthDist*faces(otherFace).n;
1042  /* Locate mean point along intersect in 3D */
1043  meanpt = minpt + mean_alongIntersect*v_intersect;
1044 
1045  /* %% Project into centroid from mean point along intersect */
1046  n = zeros(3,2);
1047  for i = 1:2
1048  /* ensure that unit normals are inward facing */
1049  n(:,i) = sign(dot(meanpt,faces(i).n))*faces(i).n;
1050  /* calculate orthogonal distance from each face to the centroid */
1051  d(i) = this.distInFromFace(CubeSat_TOF,faces(i).faceIndex);
1052  end
1053  /* Project to centroid */
1054  CubeSat_TOF.centroid_TCF= meanpt + sum(d.*n,2);
1055  }
1083  function CubeSat_TOF = centroid3(CubeSat_TOF) {
1084  CubeSat_TOF.faces= this.getFaceCentroidAndCorners(CubeSat_TOF.faces);
1085  CubeSat_TOF = this.diagnoseFace(CubeSat_TOF);
1086 
1087  /* %% Extract data */
1088  faces = CubeSat_TOF.faces;
1089 
1090  /* %% Find plane intersection */
1091  A = zeros(3);
1092  b = zeros(3,1);
1093  for i = 1:3
1094  A(i,1:3) = faces(i).n;
1095  b(i,1) = faces(i).n(1)*faces(i).o(1) + faces(i).n(2)*faces(i).o(2) + faces(i).n(3)*faces(i).o(3);
1096  end
1097  if rank(A) ~= 3
1098  error(" Planes do not intersect in the expected way ")
1099  end
1100  ptIntersect = (A\b);
1101 
1102  /* %% Project into centroid from intersection point */
1103  n = zeros(3,3);
1104  d = zeros(1,3);
1105  for i = 1:3
1106  /* ensure that unit normals are inward facing */
1107  n(:,i) = sign(dot(ptIntersect,faces(i).n))*faces(i).n;
1108  /* calculate orthogonal distance from each face to the centroid */
1109  d(i) = this.distInFromFace(CubeSat_TOF,faces(i).faceIndex);
1110  end
1111  /* Orthogonalize n */
1112  [Q,R] = qr(n);
1113  n_orth = sign(diag(R))^t .* Q;
1114  /* Project to centroid */
1115  CubeSat_TOF.centroid_TCF= ptIntersect + sum(d.*n_orth,2);
1116  }
1144  function faces = getFaceCentroidAndCorners(faces) {
1145  for ii = 1:length(faces)
1146  face = faces(ii);
1147  /* %% Project points onto 2d plane */
1148  plane = face.planeCloud;
1149  inPlane = double( (plane.Location-face(1).o^t)*face.V );
1150 
1151  /* %% Find plane boundary points */
1152  I_bound = boundary(inPlane(:,1),inPlane(:,2));
1153 
1154  /* %% Find filtered boundary */
1155  boundaryPtsRaw = [inPlane(I_bound,1)" ;inPlane(I_bound,2) "];
1156  order = 2;
1157  frameLen = this.roundToNearestOdd(size(boundaryPtsRaw,2)/5);
1158  if frameLen <= 1
1159  boundaryPts = boundaryPtsRaw;
1160  else
1161  boundaryPts = sgolayfilt(boundaryPtsRaw,order,frameLen,[],2);
1162  end
1163 
1164  /* %% Find face centroid */
1165  faceShape = polyshape(inPlane(I_bound,1),inPlane(I_bound,2)," simplify ",false);
1166  [x,y] = centroid(faceShape);
1167  face.faceCentr= face.o + face.V * [x;y];
1168 
1169  /* Find face boundary in face plane */
1170  corners_inPlane = this.minBoundingBox(boundaryPts);
1171  /* Convert to 3D */
1172  face.corners= zeros(3,4);
1173  for i = 1:4
1174  face.corners(:,i) = face.o + face.V * corners_inPlane(:,i);
1175  end
1176 
1177  /* %% repackage into faces */
1178  faces(ii) = face;
1179  end
1180  }
1201  function CubeSat_TOF = diagnoseFace(CubeSat_TOF) {
1202  for ii = 1:length(CubeSat_TOF.faces)
1203  /* Extract data */
1204  face = CubeSat_TOF.faces(ii);
1205  corners = face.corners;
1206  /* Initialize boolean */
1207  face.fullFace= 1;
1208  /* %% Decide if full face is present or not
1209  * Determine if bounding box is near cubesat size */
1210  len = zeros(1,2);
1211  deltaLen = zeros(3,2);
1212  faceMatches = cell(1,2);
1213  for i = 1:2
1214  len(i) = norm(corners(:,i+1) - corners(:,i));
1215  deltaLen(:,i) = abs(len(i) - CubeSat_TOF.actualDims);
1216  faceMatches[i] = find(deltaLen(:,i) < this.faceLenMatchTol);
1217  end
1218  /* For fullFace, both sides must have a length match */
1219  fullFaceCondition = ~isempty(faceMatches[1]) & ~isempty(faceMatches[2]);
1220  /* and each side must have an independent match */
1221  matchCombinations = combvec(faceMatches[1]" ,faceMatches{2} ");
1222  diffSides = ~(matchCombinations(1,:)-matchCombinations(2,:));
1223  fullFaceCondition = fullFaceCondition & any(diffSides);
1224 
1225  /* fullFaceCondition = fullFaceCondition & sum(ismember(faceMatches{1},faceMatches{2}))~=length(faceMatches{1});
1226  *fullFaceCondition = fullFaceCondition & sum(ismember(faceMatches{2},faceMatches{1}))~=length(faceMatches{2}); */
1227  if ~fullFaceCondition
1228  face.fullFace= 0;
1229  end
1230 
1231  /* %% Determine trusted lengths */
1232  if face.fullFace
1233  /* trusted side lengths are visible side lengths */
1234  face.trustedLen= len;
1235  /* trusted corners are identified corners */
1236  face.trustedCorners= corners;
1237  else
1238  /* NOTE: this assumes that the CubeSat has not tumbled much
1239  * since launching, which is very valid for the TOF camera
1240  * Determine trusted lengths */
1241  [trustedActualDimI,trustedLenI] = find(min(min(deltaLen))==deltaLen,1);
1242  face.trustedLen= zeros(1,2);
1243  face.trustedLen(1) = len(trustedLenI);
1244  if trustedActualDimI ~= find(CubeSat_TOF.actualDims == max(CubeSat_TOF.actualDims))
1245  face.trustedLen(2) = max(CubeSat_TOF.actualDims);
1246  else
1247  face.trustedLen(2) = min(CubeSat_TOF.actualDims);
1248  end
1249  /* %% Determine trusted corners
1250  * project to centroid from two most distant (downrange) corners
1251  * Obtain distant two corners and their midPt */
1252  if (mean(corners(3,:)) > 3) || (CubeSat_TOF.rangeOrder == 1)
1253  sortDirection = " ascend ";
1254  else
1255  sortDirection = " descend ";
1256  end
1257  [~,I] = sort(face.corners(3,:)," descend ");
1258  farCorners = face.corners(:,I(1:2));
1259  nearBadCornersUnordered = face.corners(:,I(3:4));
1260 
1261  /* Determine which trustedLen corresponds to the side that
1262  * is orthogonal to corners */
1263  [~,I] = min(abs(face.trustedLen-norm(farCorners(:,1)-farCorners(:,2))));
1264  projLength = face.trustedLen([1:I-1,I+1:end]);
1265 
1266  /* Determine near bad corners which correspond to far corners */
1267  nearBadCorners = zeros(3,2);
1268  for i = 1:size(farCorners,2)
1269  for j = 1:size(nearBadCornersUnordered,2)
1270  v1 = farCorners(:,[1:i-1,i+1:end]) - farCorners(:,i);
1271  v2 = nearBadCornersUnordered(:,j) - farCorners(:,i);
1272  if abs(dot(v1,v2)) < 1e-06
1273  nearBadCorners(:,i) = nearBadCornersUnordered(:,j);
1274  end
1275  end
1276  end
1277 
1278  /* Project through near bad corners to near corners */
1279  nearCorners = zeros(3,2);
1280  for i = 1:2
1281  u = this.unitvec(nearBadCorners(:,i) - farCorners(:,i));
1282  nearCorners(:,i) = farCorners(:,i) + projLength*u;
1283  end
1284 
1285  /* Assemble corners in circular order */
1286  face.trustedCorners= [fliplr(nearCorners),farCorners];
1287  end
1288 
1289  /* %% Determine faceIndex
1290  * Dimensions to choose from, corresponding to indices of
1291  * CubeSat.actualDims */
1292  dims = 1:length(CubeSat_TOF.actualDims);
1293  /* Calculate standard areas */
1294  C = combnk(dims,2);
1295  Avec = prod(CubeSat_TOF.actualDims(C),2);
1296  /* Calculate difference between area and standard areas */
1297  faceArea = prod(face.trustedLen);
1298  dA = abs(faceArea-Avec);
1299  /* Choose nearest size */
1300  [~,face.faceIndex] = min(dA);
1301 
1302  /* %% Repackage into CubeSat */
1303  CubeSat_TOF.faces(ii) = face;
1304  end
1305 
1306  }

1536  function bb = minBoundingBox(X) {
1537 
1538  /* compute the convex hull (CH is a 2*k matrix subset of X) */
1539  k = convhull(X(1,:),X(2,:));
1540  CH = X(:,k);
1541 
1542  /* compute the angle to test, which are the angle of the CH edges as:
1543  * "one side of the bounding box contains an edge of the convex hull" */
1544  E = diff(CH,1,2); /* CH edges */
1545 
1546  T = atan2(E(2,:),E(1,:)); /* angle of CH edges (used for rotation) */
